LLMChain

  • Chain to run queries against LLMs
    • components: Prompt + LLM + OutputParser
    • Deprecated since 0.1.17
# Use below instead of LLMChain (Deprecated)
from langchain_core.output_parsers import StrOutputParser
from langchain_core.prompts import PromptTemplate
from langchain_openai import OpenAI

prompt_template = "Tell me a {adjective} joke"
prompt = PromptTemplate(
    input_variables=["adjective"], template=prompt_template
)
llm = OpenAI()
chain = prompt | llm | StrOutputParser()

chain.invoke("your adjective here")

SequentialChain

  • Simple chain where the output of one step feed directly into next.
    • Sequential chain does not support any kind of branching or conditioning
    • just an extension of simple LLMChain where the output of the current component serves as an input to the next component
chain = prompt1 | llm2 | parser1 | prompt2 | llm2 | parser2

ParallelChain

  • A type of chain that allows parallel running of more than one sub-chains.
    • Each sub-chain has its own input and output.
    • All sub-chains are executed simultaneously.
    • Wait until all sub-chains are executed completely
    • Combine the output of all sub-chains.
from langchain_core.runnables import RunnableParallel

parallel_chain = RunnableParallel({
    "facts": fact_prompt | llm | parser,
    "definition": definition_prompt | llm | parser,
    "use_case": usecase_prompt | llm | parser
})
result = parallel_chain.invoke({"topic": "AI"})

ConditionalChain

  • Depending on if the condition is satisfied, we execute the sub-chain

def is_long_text(input: dict) -> bool:
    return len(input["text"].split()) > 50

summary_chain = summary_prompt | llm | parser
extend_chain = extend_prompt | llm | parser

conditional_chain = RunnableBranch(
    (lambda x : is_long_text(x), summary_chain),
    extend_chain
)
